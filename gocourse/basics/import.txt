package main

import (
	"fmt"
	"net/http"
)
// tree shaking in Go is handled by the compiler and linker automatically. Unused imports are removed during the build process.

func main() {
	fmt.Println("Starting server on :8080")
	resp, err := http.Get("http://example.com")
	if err != nil {
		fmt.Println("Error fetching URL:", err)
		return
	}
	defer resp.Body.Close()
	fmt.Println("Response status:", resp.Status)
}
